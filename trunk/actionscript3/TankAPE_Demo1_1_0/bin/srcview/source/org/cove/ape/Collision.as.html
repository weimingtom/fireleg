package org.cove.ape {

    import flash.utils.*;
    
    public final class Collision implements IExternalizable {
    
        internal var vn:Vector;
        internal var vt:Vector;
        
        public function Collision(vn:Vector = null, vt:Vector = null) {
            this.vn = vn;
            this.vt = vt;
        }
        
        public function writeExternal(output:IDataOutput):void {
            output.writeObject(vn);
            output.writeObject(vt);
        }
          
        public function readExternal(input:IDataInput):void {  
            vn = input.readObject() as Vector;
            vt = input.readObject() as Vector;
        }        
    }
}
